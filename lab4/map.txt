<script src="https://d3js.org/d3.v7.min.js"></script>
<script src="https://unpkg.com/topojson@3"></script>
<script>
var width = 960,
    height = 500;

var projection = d3.geoMercator()
    .center([0, 5 ])
    .scale(150)
    .rotate([-180,0]);

var svg = d3.select("body").append("svg")
    .attr("width", width)
    .attr("height", height);

var path = d3.geoPath()
    .projection(projection);

var g = svg.append("g");

// load and display the World
d3.json("https://raw.githubusercontent.com/deldersveld/topojson/master/world-countries.json").then(function(topology) {
console.log("topology ",topology);
   var topologypath = g.selectAll("path")
       .data(topojson.feature(topology, topology.objects.countries1).features)
       .enter().append("path")
       .attr("d", path)
       .on("mouseover",function(d,i){
           console.log(d,i);
       });
       console.log("topologypath :::",topologypath)

});

</script>
</body>
