     <script src='https://d3js.org/d3.v7.min.js'></script>
      <script>
         var csv1 = 'https://raw.githubusercontent.com/rahoofhw/F21DV/main/lab1/files/chartdata.csv';
         var csv2 = 'https://raw.githubusercontent.com/rahoofhw/F21DV/main/lab1/files/chartdata2.csv';

         plotbarchart(csv1);
         plotbarchart(csv2);

         function plotbarchart(csv){
            var svg = d3.select("body").append("svg").attr("width", 510).attr("height", 200);     
            const width = 500;
            const barHeight = 20;
            const margin = 1;
               
            d3.csv(csv, function(data,i) {
               var d= data.value;
               var color = 'blue';
               
               if(d>500){color= 'red';} else if(d<100){color= 'green';}
               
               var g = svg.append("g") .attr("transform","translate(0," + i * barHeight + ")");
               g.append("rect").attr("width", d/2).attr("height", barHeight - margin).attr('fill', color);
               g.append("text").attr("x", d/2).attr("y", barHeight / 2).attr("dy", ".35em").style('text-anchor', 'end').text(d);
            });
         }
      </script>
