     <script src='https://d3js.org/d3.v7.min.js'></script>
      <script>
         var csv1 = 'https://raw.githubusercontent.com/rahoofhw/F21DV/main/lab1/files/chartdata.csv';
         var svg = d3.select("#canvas").append("svg").attr("width", 400).attr("height", 400);     
         const width = 500;
         const barHeight = 20;
         const margin = 1;
            
         d3.csv(csv1, function(data,i) {
            var d= data.value;
            var color = 'blue';
            
            if(d>500){color= 'red';} 
            else if(d<100){color= 'green';}
            
            var g = svg.append("g") .attr("transform","translate(0," + i * barHeight + ")");
            g.append("rect").attr("width", d).attr("height", barHeight - margin).attr('fill', color);
            g.append("text").attr("x", d).attr("y", barHeight / 2).attr("dy", ".35em").style('text-anchor', 'end').text(d);
         });
      </script>
